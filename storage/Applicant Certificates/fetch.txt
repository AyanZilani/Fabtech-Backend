import React, { Component } from 'react';
import axios from 'axios';
import Table from './Table.js';


class Job_available extends Component{

    constructor(props) {
        super(props);
        this.state = {
            droplets: []
        }
    }   

    componentDidMount() {
        fetch('http://localhost:3000/api/Jobs_availables')
        .then(res => res.json())
        .then(json => json.droplets)
        .then(droplets => this.setState({ 'droplets': droplets }))
      }

    
    render(){
            return (         
                <div>
                <Table droplets={ this.state.droplets } />
              </div>  
      )
    }
}

export default Job_available
